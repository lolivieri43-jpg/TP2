TP8 : LES TABLEAUX


Introduction

Dans ce TP, il est demandé d'écrire des fonctions utilitaires permettant de travailler avec les tableaux. Ces fonctions seront réutilisables dans d'autres TP.

Le tableau contient N éléments. Il est interdit d'ajouter, d'insérer ou de supprimer des cases. 

let N = parseInt(prompt("Combien de case pour le tableau ?"));
let A = new Array(N);
let nbElemRempli = 0; // le tableau contient N cases vides

Aperçu d’un tableau vide :

Index : 0   1   2   3   ...   N-2   N-1
Valeur: undefined undefined undefined ...

Méthode de travail

  1. Boîte noire : définir les entrées et sorties
  2. Coder la fonction dans un fichier .js séparé
  3. Ajouter le mode d’emploi
  4. Tester avec l’inspecteur du navigateur

Exercices

Exercice 1

initialiserTab1() : initialise toutes les cases à 0.

Exercice 2

initialiserTab2(valeur) : initialise toutes les cases avec une valeur passée en paramètre.

Exercice 3

remplirTab1(min, max) : remplit tout le tableau avec des valeurs aléatoires entre MIN et MAX.

Exercice 4

remplirTab2(nb, min, max) : remplit un certain nombre de cases à partir de la première case vide (nbElemRempli). Si dépassement, on remplit jusqu'à N. 

Exemple :

Index : 0   1   2   3   4   5   6   7
Valeur: 5   12  3   undefined ...
nbElemRempli = 3

Exercice 5

afficherTab1() : affiche uniquement les cases remplies.

Exercice 6

afficherTab2() : affiche toutes les cases.

Exercice 7

afficherTab3(deb, fin) : affiche les cases entre DEB et FIN.

Exercice 8

remplirElement(valeur) : insère une valeur dans la première case vide. Retourne false si le tableau est plein, sinon retourne le nombre d’éléments remplis.

Exercice 9

saisirTab() : permet à l’utilisateur de saisir des valeurs ajoutées à partir de la première case vide. Retourne le nombre de cases remplies.

Fichier utilitaires.js


function initialiserTab1(A) {
    for (let i = 0; i < A.length; i++) {
        A[i] = 0;
    }
    return A.length;
}

function initialiserTab2(A, val) {
    for (let i = 0; i < A.length; i++) {
        A[i] = val;
    }
    return A.length;
}

function remplirTab1(A, MIN, MAX) {
    for (let i = 0; i < A.length; i++) {
        A[i] = Math.floor(Math.random() * (MAX - MIN + 1)) + MIN;
    }
    return A.length;
}

function remplirTab2(A, nbElemRempli, nbARemplir, MIN, MAX) {
    let i = nbElemRempli;
    let fin = Math.min(A.length, nbElemRempli + nbARemplir);
    while (i < fin) {
        A[i] = Math.floor(Math.random() * (MAX - MIN + 1)) + MIN;
        i++;
    }
    return i;
}

function afficherTab1(A, nbElemRempli) {
    let res = "";
    for (let i = 0; i < nbElemRempli; i++) {
        res += A[i] + " ";
    }
    console.log(res);
}

function afficherTab2(A) {
    let res = "";
    for (let i = 0; i < A.length; i++) {
        res += A[i] + " ";
    }
    console.log(res);
}

function afficherTab3(A, DEB, FIN) {
    let res = "";
    for (let i = DEB; i <= FIN && i < A.length; i++) {
        res += A[i] + " ";
    }
    console.log(res);
}

function remplirElement(A, nbElemRempli, val) {
    if (nbElemRempli >= A.length) {
        return false;
    }
    A[nbElemRempli] = val;
    return nbElemRempli + 1;
}

function saisirTab(A, nbElemRempli) {
    let nb = parseInt(prompt("Combien de valeurs voulez-vous saisir ?"));
    let i = 0;
    while (i < nb && nbElemRempli < A.length) {
        let val = parseInt(prompt("Donnez une valeur :"));
        A[nbElemRempli] = val;
        nbElemRempli++;
        i++;
    }
    return nbElemRempli;
}
